---
title: "Query Process Runs"
api: "GET https://artemisarbackend.com:8080/ext/query_process_runs"
description: "This endpoint queries and returns a specified run of a specified process"
---

### Body

<ParamField body="api_key" type="string" required>
  This is the API key for your account. You can obtain this from the "Secrets" tab of your "Settings" page.
</ParamField>

<ParamField body="subproject_id" type="string" required>
  This is the id of the subproject the process resides in. You can obtain this from the URL of the subproject page.
</ParamField>

<ParamField body="process_id" type="string" required>
    This is the id of the process whose runs you are trying to query. You can obtain this from the table on the first tab of the process modal.
</ParamField>

### Response

<ResponseField name="status" type="number">
  Indicates whether the call was successful. 200 if successful, 400 if not.
</ResponseField>

<ResponseField name="message" type="[Process Run]">
  Message containing the queried process run.
  <Expandable title="Process Run Object Structure">
    <ResponseField name="name" type="string" required>
        This is the name of the process run
    </ResponseField>
    <ResponseField name="creator" type="string" optional>
        The name of the user who created the process run
    </ResponseField>
    <ResponseField name="run_date" type="string" optional>
        This is the date the process run was run (MM/DD/YYYY)
    </ResponseField>
    <ResponseField name="id" type="string" optional>
        This is the id of the process run
    </ResponseField>
  </Expandable>
</ResponseField>

--header 'Content-Type: application/json' \
--header 'Authorization: Token <token>' \
