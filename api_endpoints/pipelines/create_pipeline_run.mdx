---
title: "Create Pipeline Run"
api: "POST https://artemisarbackend.com:8080/ext/create_pipeline_run"
description: "This endpoint creates a run of a specified pipeline"
---

### Body

<ParamField body="api_key" type="string" required>
  This is the API key for your account. You can obtain this from the "Secrets" tab of your "Settings" page.
</ParamField>

<ParamField body="project" type="string" required>
  This is the id of the project the pipeline run will reside in. You can obtain this from the URL of the project page.
</ParamField>

<ParamField body="pipeline_id" type="string" required>
    This is the id of the pipeline to be run. You can obtain this from the table on the first tab of the pipeline modal.
</ParamField>

<ParamField body="pipeline_version" type="string" required>
    This is the version of the pipeline associated with this run. This is the name of the version indicated in the verison dropdown. The default version is "Version 1".
</ParamField>

<ParamField body="name" type="string" required>
  This is the name of the pipeline run to be created.
</ParamField>

<ParamField body="description" type="string" required>
    This is the description of the pipeline run to be created.
</ParamField>

<ParamField body="run_date" type="string" optional>
    This is the date the pipeline run was run (MM/DD/YYYY). If not provided, the current date will be used.
</ParamField>

<ParamField body="steps" type="[Steps]" optional>
  An array of steps indicating the information to be logged for each step in the pipeline for this run.
   <Expandable title="Step Structure">
    <ResponseField name="name" type="string" required>
        The name of the step. This can be found by double clicking the step in the network
    </ResponseField>
    <ResponseField name="description" type="string" optional>
        The description of the step
    </ResponseField>
     <ResponseField name="input_notes" type="string" optional>
        Any notes about the input for the step
    </ResponseField>
     <ResponseField name="output_notes" type="string" optional>
        Any notes about the output for the step
    </ResponseField>
    <ResponseField name="origin_notes" type="string" optional>
        Any notes about the origin of the step
    </ResponseField>
    <ResponseField name="deployment_notes" type="string" optional>
        Any notes about the deployment of the step
    </ResponseField>
    <ResponseField name="fields" type="[Partial Field]" optional>
        An array of partial fields indicating which fields you want to log for this step and the values they should be updated to. Note: You only need to specify the fields you want to log values for.
        <Expandable title="Partial Field Structure">
            <ResponseField name="name" type="string" required>
                The name of the field
            </ResponseField>
            <ResponseField name="value" type="type" optional default="">
                The updated value for this field
            </ResponseField>
        </Expandable>
    </ResponseField>
    <ResponseField name="underlying_resource_content" type="[Content]" optional>
        **Note: This is only applicable to dynamic steps.** This is an array of content items indicating the content of the underlying resource for this step. The structure of each Content item is detailed below.
         <Expandable title="Content Structure">
            <ResponseField name="name" type="string" required>
            The name of the content
            </ResponseField>
            <ResponseField name="version" type="string" optional>
                The version of the content
            </ResponseField>
            <ResponseField name="location" type="string" optional>
                The location of the content
            </ResponseField>
            <ResponseField name="description" type="string" optional>
                The description of the content
            </ResponseField>
        </Expandable>
    </ResponseField>
    <ResponseField name="underlying_resource_description" type="string" optional>
        **Note: This is only applicable to dynamic steps.**  This is the description of the underlying resource for this step.
    </ResponseField>
    <ResponseField name="underlying_resource_content_notes" type="string" optional>
        **Note: This is only applicable to dynamic steps.**  This is the notes of the content of the underlying resource for this step.
    </ResponseField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="status" type="number">
  Indicates whether the call was successful. 200 if successful, 400 if not.
</ResponseField>

<ResponseField name="message" type="string">
  Message saying "Pipeline run created successfuly" if successful, an error message if not.
</ResponseField>
